export const styleImage = (tipo, n, id) => {
  switch (tipo) {
    case "1":
      return html`
        <img alt="${n}" src="https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/home/${id}.png">
      `;
    case "2":
      return html`
        <img alt="${n}" src="https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/${id}.png">
      `;
    case "3":
      return html`
       <img alt="${n}" style="filter: saturate(1.6)" src="https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/official-artwork/${id}.png">
      `;
    case "4":
      return html`
       <img alt="${n}" src="https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/dream-world//${id}.svg">
      `;
    case "5":
      return html`
       <img alt="${n}" style="width: 30%; margin: 0 auto;" src="https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/showdown/${id}.gif">
      `;
  
    default:
      return html`
        <img alt="${n}" src="https://raw.githubusercontent.com/PokeAPI/sprites/master/sprites/pokemon/other/home/${id}.png">
      `;
  }
 
};
